// 安装插件
app.use(pluginTheme);
app.use(pluginLogger);
app.use(pluginAuth);
app.use(pluginStorage);
app.use(pluginI18n);
app.use(pluginChart);

// 使用主题插件
app.config.globalProperties.$theme.setTheme('dark');

// 使用日志插件
app.config.globalProperties.$logger.info('Hello World');

// 使用认证插件
app.config.globalProperties.$auth.login(user);

// 使用存储插件
app.config.globalProperties.$storage.set('key', 'value');

// 使用国际化插件
app.config.globalProperties.$t('greeting', { name: 'John' });

// 使用图表插件
const chart = app.config.globalProperties.$chart.create(canvas, config);
