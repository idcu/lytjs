🎨 LytJS UI 组件库

零第三方依赖，支持主题切换的现代化 UI 组件库

🔘 Button 按钮

支持多种类型和尺寸的按钮组件

import { Button } from '@lytjs/ui' <Button type="primary"> 点击我 </Button>

📝 Input 输入框

支持多种类型的表单输入框

import { Input } from '@lytjs/ui' <Input placeholder="请输入内容" />

📋 Table 表格

功能强大的表格组件，支持排序、选择、固定列

import { Table } from '@lytjs/ui' <Table data={data} columns={cols} />

🌳 Tree 树形控件

树形选择器，支持拖拽、异步加载

import { Tree } from '@lytjs/ui' <Tree data={treeData} />

📑 Modal 弹窗

对话框组件，支持拖拽和全屏显示

import { Modal } from '@lytjs/ui' <Modal title="标题" visible={true}> 内容 </Modal>

📬 Notification 通知

消息提示组件，支持多种位置和类型

import { Notification } from '@lytjs/ui' Notification.success('操作成功')

🏷️ Tag 标签

标签组件，可用于分类和筛选

import { Tag } from '@lytjs/ui' <Tag type="success">完成</Tag>

🎭 Badge 徽标

用于显示数字或状态的徽标组件

import { Badge } from '@lytjs/ui' <Badge count={5}>消息</Badge>

⏰ DatePicker 日期选择

日期选择器，支持范围选择

import { DatePicker } from '@lytjs/ui' <DatePicker />

📤 Upload 文件上传

文件上传组件，支持分片上传

import { Upload } from '@lytjs/ui' <Upload action="/upload" />

📇 Drawer 抽屉

侧边抽屉组件

import { Drawer } from '@lytjs/ui' <Drawer title="菜单" />

📊 Descriptions 描述列表

详情描述组件

import { Descriptions } from '@lytjs/ui' <Descriptions title="用户信息" />

☑️ CheckboxGroup 复选框组

多选复选框组件，支持多选和禁用

import { CheckboxGroup, Checkbox } from '@lytjs/ui' <CheckboxGroup modelValue={['选项1']}> <Checkbox label="选项1" /> <Checkbox label="选项2" /> </CheckboxGroup>

🔘 RadioGroup 单选框组

单选框组件，支持单选和禁用

import { RadioGroup, Radio } from '@lytjs/ui' <RadioGroup modelValue="选项1"> <Radio label="选项1" /> <Radio label="选项2" /> </RadioGroup>

📊 Progress 进度条

进度条组件，支持线型和圆形

import { Progress } from '@lytjs/ui' <Progress percentage={60} /> <Progress type="circle" percentage={75} />

🎚️ Slider 滑块

滑块组件，支持单值和范围选择

import { Slider } from '@lytjs/ui' <Slider modelValue={50} /> <Slider range modelValue={[20, 80]} />

👤 Avatar 头像

头像组件，支持图片、图标和文字

import { Avatar } from '@lytjs/ui' <Avatar src="https://example.com/avatar.jpg" /> <Avatar icon="user" /> <Avatar>用户</Avatar>

📇 Card 卡片

卡片组件，支持阴影和标题

import { Card } from '@lytjs/ui' <Card title="卡片标题"> 卡片内容 </Card>

📅 Timeline 时间轴

时间轴组件，用于展示事件顺序

import { Timeline, TimelineItem } from '@lytjs/ui' <Timeline> <TimelineItem timestamp="2026-05-14"> 事件内容 </TimelineItem> <TimelineItem type="success">完成</TimelineItem> </Timeline>

📊 Steps 步骤条

步骤条组件，用于引导操作流程

import { Steps, Step } from '@lytjs/ui' <Steps active={1}> <Step title="步骤一" description="描述" /> <Step title="步骤二" status="process" /> <Step title="步骤三" /> </Steps>

🎠 Carousel 走马灯

走马灯组件，用于轮播展示内容

import { Carousel, CarouselItem } from '@lytjs/ui' <Carousel height="300px" autoplay> <CarouselItem>内容1</CarouselItem> <CarouselItem>内容2</CarouselItem> </Carousel>

💬 Popconfirm 气泡确认框

气泡确认框组件，用于操作前确认

import { Popconfirm, Button } from '@lytjs/ui' <Popconfirm title="确定要删除吗？" onConfirm={handleConfirm}> <Button>删除</Button> </Popconfirm>
